    <!-- JavaScript Functions -->
    <script>
        // පාරිභෝගිකයා ලොග් වී ඇත්දැයි බැලීමට (Simulated User State)
        let isLoggedIn = false;
        let loggedInUser = "";

        const modal = document.getElementById('authModal');
        const loginSection = document.getElementById('loginFormSection');
        const registerSection = document.getElementById('registerFormSection');
        const navAuthBtn = document.getElementById('navAuthBtn');

        // Pop-up එක Open කිරීමට
        function openModal(type) {
            modal.style.display = 'flex';
            toggleForm(type);
        }

        // Pop-up එක Close කිරීමට
        function closeModal() {
            modal.style.display = 'none';
        }

        // Login සහ Register Form මාරු කිරීමට
        function toggleForm(type) {
            if(type === 'login') {
                loginSection.style.display = 'block';
                registerSection.style.display = 'none';
            } else {
                loginSection.style.display = 'none';
                registerSection.style.display = 'block';
            }
        }

        // Register සිදුවන අවස්ථාව
        function handleRegister(e) {
            e.preventDefault();
            alert("Registration Successful! Now you can Login.");
            toggleForm('login');
        }

        // Login සිදුවන අවස්ථාව
        function handleLogin(e) {
            e.preventDefault();
            isLoggedIn = true;
            alert("Login Successful! You can proceed to buy.");
            navAuthBtn.innerText = "Logged In";
            navAuthBtn.style.backgroundColor = "#2ed573";
            closeModal();
        }

        // Buy Now Button එක Click කල විට සිදුවන දේ
        function handleBuyNow(productName) {
            if (!isLoggedIn) {
                alert("Please Register or Login first to purchase '" + productName + "'!");
                openModal('register'); // ලොග් වී නැත්නම් කෙලින්ම Register Form එක පෙන්වයි
            } else {
                alert("Thank you! Processing your order for: " + productName);
                // මෙතනින් ඉදිරියට Payment Gateway එකකට හෝ Checkout පිටුවකට යා හැක.
            }
        }

        // පැත්තක් Click කලොත් Pop-up එක වැසීමට
        window.onclick = function(event) {
            if (event.target == modal) {
                closeModal();
            }
        }
    </script>
